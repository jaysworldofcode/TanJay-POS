CREATE DATABASE pos_tanjay

//Create stocks table
CREATE TABLE STOCKS_TABLE(STOCK_ID int AUTO_INCREMENT, BARCODE char(255), DATE_RECEIVED varchar(255), DATE_EXPIRY varchar(255), ITEM_NAME varchar(255), PRICE varchar(255), UNIT_LEFT int, EMPTY_EXPIRY varchar(255), PRIMARY KEY (STOCK_ID))

//Create credit log
CREATE TABLE CREDIT_LOG(CREDIT_ID int AUTO_INCREMENT, CUSTOMER_ID char(255), ITEM_NAME varchar(255), CODE varchar(255), PRICE varchar(255), DATE_ISSUED varchar(255), UNIT int, STATUS varchar(255), CASHIER varchar(255), DESCRIPTION varchar(255), PRIMARY KEY (CREDIT_ID))

//Create customer table
CREATE TABLE CUSTOMER_TABLE(CUSTOMER_ID int AUTO_INCREMENT, CUSTOMER_NAME char(255), CUSTOMER_ADDRESS varchar(255), CUSTOMER_CONTACT_NUMBER varchar(255), PRIMARY KEY (CUSTOMER_ID))
